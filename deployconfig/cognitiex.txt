upstream springboot {
    server 127.0.0.1:8080;
}

upstream gunicorn {
    server unix:/run/gunicorn.sock fail_timeout=0;
}

server {
    listen 80;
    server_name cognitiex.com www.cognitiex.com;
    location /.well-known/acme-challenge/ { root /var/www/certbot; }
    location / { return 301 https://$server_name$request_uri; }
}

server {
    listen 443 ssl http2;
    server_name cognitiex.com www.cognitiex.com;
    ssl_certificate /etc/letsencrypt/live/cognitiex.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/cognitiex.com/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
    access_log /var/log/nginx/cognitiex.log combined buffer=32k;
    error_log /var/log/nginx/cognitiex-error.log warn;
    client_max_body_size 100M;
    root /var/www/ssqqq/dist/3-dproject/browser;
    index index.csr.html;
    gzip on;
    gzip_types text/plain text/css application/json application/javascript;
    gzip_min_length 1000;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-Frame-Options "SAMEORIGIN" always;

    location /static/ {
        alias /var/www/ssqqq/assistance/static/;
        expires 30d;
    }

    location /media/ {
        alias /var/www/ssqqq/assistance/media/;
        expires 7d;
    }

    location = /robots.txt {
        try_files $uri /index.csr.html =404;
    }

    location /uploads/ {
        alias /var/www/ssqqq/School/src/main/resources/static/uploads/;
        expires 30d;
    }

    location /api/ {
        proxy_pass http://springboot;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    location /ai/ {
        proxy_pass http://gunicorn;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_buffering off;
        proxy_connect_timeout 60s;
        proxy_send_timeout 120s;
        proxy_read_timeout 120s;
    }

        # ===== ANGULAR SPA =====

    location = /index.csr.html {
        add_header Cache-Control "no-cache, no-store, must-revalidate";
    }

    location / {
        try_files $uri /index.csr.html;
        add_header Cache-Control "must-revalidate";
    }

    location ~ /\. {
        deny all;
    }
}
